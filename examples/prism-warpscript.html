//factorial macro. take a number on the stack, push its factorial
<%
	'input' STORE
	1
	1 $input <% * %> FOR
%> 'factorial' STORE

//build a map with key from 1 to 10 and value = key!
{} 'result' STORE

1 10
<%
	'key' STORE
	$result $key @factorial $key PUT
	DROP //remove the map let by PUT
%> FOR

//push the result on the stack
$result
