partial def String.isSubStrOf (sub : String) (s : String) : Bool :=
  loop 0
  where
    loop i :=
	  if i + sub.length > s.length then false
	  else String.substrEq sub 0 s ⟨i⟩ sub.length \/ loop (i + 1)
